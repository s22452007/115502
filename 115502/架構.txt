=======================================================================
📱 J-Lens: 日語情境學習 App - 完整程式碼架構與功能說明
=======================================================================

根目錄/
├── backend/                             [後端：Python Flask 伺服器]
│   ├── app.py                           # 伺服器啟動入口：註冊所有 Blueprint 模組與初始化環境
│   ├── models.py                        # 資料庫綱要：定義 User, Vocab, UserPhoto, UserVocab 等實體關係
│   ├── seed.py                          # 假資料腳本：一鍵清除並重建開發測試用的資料庫內容
│   ├── upgrade_db.py                    # 資料庫工具：用於安全新增欄位或修改現有表結構
│   ├── requirements.txt                 # 套件清單：列出 Flask, SQLAlchemy 等依賴套件
│   ├── .env                             # 環境變數：存放 API 金鑰、Secret Key 等機密資訊
│   ├── instance/                        # 資料庫資料夾：存放 SQLite 實體檔案 jlens.db
│   ├── static/photos/                   # 靜態資源：存放使用者上傳的場景照片實體檔
│   ├── utils/                           # 核心工具箱
│   │   ├── ai_helper.py                 # 封裝 OpenAI/Gemini API，處理圖片辨識與對話生成
│   │   ├── jwt_helper.py                # 處理 JWT Token 的產生、加密與身份驗證邏輯
│   │   ├── auth_helper.py               # 授權與 ID 生成相關的小工具
│   │   └── group_helper.py              # 計算學習小組進度與排行的邏輯
│   └── services/                        # 控制器層 (Blueprint 路由定義)
│       ├── auth.py                      # 處理註冊、登入、密碼重設與 Google 驗證
│       ├── user.py                      # 處理個人資料、能力值、點數與徽章進度抓取
│       ├── scenario.py                  # 處理 AI 拍照辨識分析與拍照歷史事件 (UserPhoto)
│       ├── vocabulary.py                # 處理單字圖鑑、收藏狀態 (UserVocab) 與資料夾管理
│       ├── roleplay.py                  # 處理 AI 角色扮演對話流與語音自然度評分
│       ├── tutor.py                     # 處理 AI 家教問答紀錄與 FAQ
│       ├── group.py                     # 處理學習小組的建立、邀請與成員管理
│       └── quiz.py                      # 處理新手能力測驗題庫與等級判定邏輯
│
└── jpn_learning_app/                    [前端：Flutter 行動應用程式]
    ├── pubspec.yaml                     # 專案設定檔：管理 Flutter 套件版本與 assets 資源路徑
    ├── .env                             # 前端環境變數：存放後端伺服器 IP 網址
    └── lib/                             # 前端核心程式碼 (Dart)
        ├── main.dart                    # App 總入口：初始化 Provider、設定路由與全域主題
        ├── firebase_options.dart        # Firebase 設定檔：自動生成的雲端服務串接金鑰
        │
        ├── models/                      # 資料模型層 (JSON 序列化)
        │   ├── user.dart                # 使用者帳號與能力資料模型
        │   ├── vocabulary.dart          # 單字解釋、音檔路徑與例句模型
        │   ├── scenario.dart            # 拍照場景事件模型
        │   ├── badge_model.dart         # 徽章名稱與達成門檻模型
        │   ├── leaderboard.dart         # 排行榜進度資料模型
        │   └── question.dart            # 測驗題庫題目格式模型
        │
        ├── providers/                   # 狀態管理層 (資料管家)
        │   ├── auth_provider.dart       # 管理登入狀態、Token 記憶與登出權限清除
        │   ├── user_provider.dart       # 核心狀態：儲存暱稱、等級、點數、徽章進度與雷達圖數值
        │   ├── scenario_provider.dart   # 暫存拍照辨識結果，供結果頁面即時讀取
        │   ├── favorites_data.dart      # 單字收藏狀態快取，優化按星星時的 UI 反應速度
        │   └── font_size_provider.dart  # 紀錄使用者自訂的字體大小倍率
        │
        ├── services/                    # API 通訊層 (外送員)
        │   ├── auth_service.dart        # 對接後端 auth 路由，傳送憑證進行認證
        │   ├── ai_service.dart          # 對接 AI 功能，上傳圖片或傳送對話文字
        │   ├── user_service.dart        # 請求個人檔案、好友列表與排行榜數據
        │   ├── vocab_service.dart       # 處理單字收藏與資料夾操作的網路請求
        │   ├── tutor_service.dart       # 傳送家教提問與獲取回覆內容
        │   ├── notification_service.dart# 處理推播通知與系統訊息提醒
        │   ├── rank_service.dart        # 獲取全球排行榜與小組排行
        │   └── api_client.dart          # HTTP 基底：統一管理 baseUrl, Header 與錯誤處理
        │
        ├── utils/                       # 前端工具類
        │   ├── constants.dart           # 定義全域調色盤 (AppColors)、文字樣式與靜態網址
        │   └── badge_utils.dart         # 封裝徽章等級計算公式 (例如：經驗值如何換算等級)
        │
        ├── screens/                     # 畫面層 (UI Screens)
        │   ├── auth/                    # 1. 認證模組
        │   │   ├── splash_screen.dart   # 閃屏：顯示 Logo 並執行啟動檢查

        │   │   ├── login_screen.dart    # 登入與註冊的主控制頁
        │   │   ├── forgot_password_screen.dart# 忘記密碼流程頁
        │   │   ├── level_select_screen.dart# 程度選擇：使用者自評 N1~N5
        │   │   ├── quick_test_screen.dart # 快速測驗：能力鑑定題目作答區
        │   │   └── test_result_screen.dart# 測驗結果：判定初始等級與落點分析
        │   │        
        │   ├── home/                    # 2. 首頁模組
        │   │   └── home_screen.dart     # 首頁：顯示每日目標、學習火種與捷徑
        │   │
        │   ├── scenario/                # 3. 核心辨識模組
